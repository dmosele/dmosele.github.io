\begin{equation*} f'(1) = \lim_{h\to 0} \frac{f(1+h)-f(1)}{h} \end{equation*}
\begin{equation*} = \lim_{h\to 0} \frac{-3(1+h)^2+7(1+h)-2-(2)}{h} \end{equation*}
\begin{equation*} = \lim_{h\to 0} \frac{-3-6h-3h^2+7+7h-4}{h} \end{equation*}
\begin{equation*} = \lim_{h\to 0} \frac{-3h^2+h}{h} \end{equation*}
\begin{equation*} = \lim_{h\to 0} -3h+1 \end{equation*}
\begin{equation*} = 1 \end{equation*}
